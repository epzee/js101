Scope


var foo = function () {
	var a = 3, b = 5;
	var bar = function () {
		var b = 7, c = 11;
		// a = 3, b = 7, and c = 11
		a += b + c;
		//a = 21, b = 7, and c = 11
	};
	//a = 3, b = 5, and c = not defined
	bar();
	//a = 21, b = 5
};

Closure


var IdGenerator = {
	
	number : 0,

	getId : function(){
		this.number += 1;
		return this.number;
	}
};
console.log(IdGenerator.getId()); // 1
console.log(IdGenerator.number); // 1
console.log(IdGenerator.getId()); // 2

Closure


var IdGenerator = function () {
	
	var number = 0;

	return {
		getId : function(){
			number += 1;
			return number;
		}
	}
}();
console.log(IdGenerator.getId()); // 1
console.log(IdGenerator.number); // undefined
console.log(IdGenerator.getId()); // 2
